% % % % Inspired by
% % % % o By S.Gibson, School of Physical Sciences, University of Kent.
% % % % o 1st May, 2013.

% % % % o NOTES: If the max number of iterations exceeds 25, error sometimes

% % %
% % % % FIDDLE WITH:
% % % % noisyness of subject selection process (below)
% % % % cs samples - parameter m (below)
% % % % biht parameter - sparsity (biht code) - lower (<64) is better
% % % % basis type (1d vs 2d) - rougly the same, preference for 2D
% % %
% % % %%%%%%%%%%%%%%%%%%%
% % % % Setup
% % %
l = 20000;
% % %
A = imread('line.png');
if ndims(A)>2
A = rgb2gray(A);
end
A = imresize(A,50/size(A,1)); % 0.3
A = imresize(A,[min(size(A)) min(size(A))]); % 0.3
A = double(A)./256;
% % %
n = size(A,1)*size(A,2);
m = 5000; % 2.5% of recommended 20k trials
% % %
% % % %%%%%%%%%%%%%%%%%%%
% % % % Real Basis
% % %
ii = 128;
b = dct2(A);
bm = zeros(n,1);
[val idx] = sort(abs(b(:)),'descend');
bm(abs(b)>val(ii)) = b(abs(b)>val(ii));

% AA = idct2(reshape(bm,sqrt(n),sqrt(n)));
% figure
% imshow(AA)
% % %
% % % %%%%%%%%%%%%%%%%%%%
% % % % Subject Selection Process
% % %
AA = repmat(reshape(A,1,n),l,1);%make matrix of the "s" image over and over
X = round(rand(l,n)); %white noise
% % %
% Ideal selection
e = sum((X-AA).^2,2);
y = double(e>=prctile(e,80.0));%yes percentage (xx.x)
y(y==0) = -1;
% % %
% % % % % % % Noisy selection
% % % % % % e = sum((X-AA).^2,2);
% % % % % % e = e + 0.2*range(e)*randn(size(e));
% % % % % % y = double(e>=prctile(e,49.5));
% % % % % % y(y==0) = -1;
% % %
% % % %%%%%%%%%%%%%%%%%%%
% % % % Linear Reconstruction a la Gosselin

x_gossl = (1/n)*X'*y;


% figure
% imagesc(reshape(x_gossl,50,50))

%%%%%%%%%%%%%%%%%%%
% Measurement Matrix
% -> a matrix of stimulus images, vectorized

Phi = X(1:m,:);

%%%%%%%%%%%%%%%%%%%
% Measurement
% -> responses from subjects, in light of stimuli (and internal model)

ym = y(1:m);

%%%%%%%%%%%%%%%%%%%
% Linear Reconstruction a la Gosselin - limited samples

x_gossm = (1/n)*Phi'*ym;

% figure
% imagesc(reshape(x_gossm,50,50))

%%%%%%%%%%%%%%%%%%%
% Basis Matrix & Theta
% -> matrix of assumed internal model basis functions
% NOTE: Avoid calculating Psi (nxn) directly to avoid memory issues.

% Theta = zeros(m,n);
% for ii = 1:n
%     ek = zeros(1,n);
%     ek(ii) = 1;
%     psi = idct(ek)';
%     Theta(:,ii) = Phi*psi;
% end

Theta = zeros(m,n);
for ii = 1:n
    ek = zeros(round(sqrt(n)),round(sqrt(n)));
    ek(ii) = 1;
    psi = reshape(idct2(ek),n,1);
    Theta(:,ii) = Phi*psi;
end

%%%%%%%%%%%%%%%%%%%
% Weighting Reconstruction - L2 Norm

s2 = pinv(Theta)*ym;

%%%%%%%%%%%%%%%%%%%
% Weighting Reconstruction - L1 Norm
%
% min_{s,u} sum(u)  s.t.  -u <= s <= u,  Theta*s=y
%
% Need: min_{s,u} sum(u)  s.t.  -u <= s <= u,  Theta*s=y

%   Basis Pursuit
%s1 = l1eq_pd(s2,Theta,Theta',y,5e-3,20); % L1-magic toolbox
%s1 = l1eq_pd(s2,Theta,Theta',y,5e-4,100); % L1-magic toolbox

%%%s1 = rfpi(s2,Theta,y);

%s1 = biht(s2,Theta,ym);

%gamma = 0.4*sqrt(log(n)/m);
%s1 = zhangpassive(Theta,ym,gamma);

gamma = 100;
s1 = zhangpassivegamma(Theta,ym,gamma);

ym_est = sign(Theta*s1);

corrcoef(ym,ym_est);

%%%%%%%%%%%%%%%%%%%
% Viz Weighting Reconstruction

% % ek = dct2(A);
% %
% % figure
% % plot(ek(:)./max(ek(:)),'b'), hold all
% % scatter(1:length(s1),s1./max(s1),'ro')
% % legend('actual dct basis','biht basis')

%%%%%%%%%%%%%%%%%%%
% IMAGE RECONSTRUCTION - LSQ

% x_csl2 = zeros(n,1);
% for ii = 1:n
%     ek = zeros(1,n);
%     ek(ii) = 1;
%     psi = idct(ek)';
%     x_csl2 = x_csl2+psi*s2(ii);
% end

% % x_csl2 = zeros(n,1);
% % for ii = 1:n
% %     ek = zeros(round(sqrt(n)),round(sqrt(n)));
% %     ek(ii) = 1;
% %     psi = reshape(idct2(ek),n,1);
% %     x_csl2 = x_csl2+psi*s2(ii);
% % end

%%%%%%%%%%%%%%%%%%%
% IMAGE RECONSTRUCTION - BIHT - Small Sample

% x_csl1 = zeros(n,1);
% for ii = 1:n
%     ek = zeros(1,n);
%     ek(ii) = 1;
%     psi = idct(ek)';
%     x_csl1 = x_csl1+psi*s1(ii);
% end

x_csl1 = zeros(n,1);
for ii = 1:n
    ek = zeros(round(sqrt(n)),round(sqrt(n)));
    ek(ii) = 1;
    psi = reshape(idct2(ek),n,1);
    x_csl1 = x_csl1+psi*s1(ii);
end

%%%%%%%%%%%%%%%%%%%
% IMAGE RECONSTRUCTION - BIHT - Full Sample

w = l;

yw = y(1:w);
Phiw = X(1:w,:);
Thetaw = zeros(w,n);
for ii = 1:n
    ek = zeros(round(sqrt(n)),round(sqrt(n)));
    ek(ii) = 1;
    psi = reshape(idct2(ek),n,1);
    Thetaw(:,ii) = Phiw*psi;
end
% % % s2w = pinv(Thetaw)*yw;
% % % s1w = biht(s2w,Thetaw,yw);
%s1w = zhangpassive(Thetaw,yw,gamma);
s1w = zhangpassivegamma(Thetaw,yw,gamma);
x_csl0 = zeros(n,1);
for ii = 1:n
    ek = zeros(round(sqrt(n)),round(sqrt(n)));
    ek(ii) = 1;
    psi = reshape(idct2(ek),n,1);
    x_csl0 = x_csl0+psi*s1w(ii);
end

% % %%%%%%%%%%%%%%%%%%%
% % % Viz Image Reconstruction - Proposal Figure
% %
figure('name','Compressive Sensing')
subplot(1,5,1), imagesc(-A), xlabel('Cognitive Representation'), axis image, set(gca,'xtick',[],'ytick',[])
subplot(1,5,2), imagesc(reshape(x_gossl,round(sqrt(n)),round(sqrt(n)))), xlabel(strcat('Conventional Estim (n=',num2str(l),')')), axis image, set(gca,'xtick',[],'ytick',[])
subplot(1,5,3), imagesc(reshape(x_gossm,round(sqrt(n)),round(sqrt(n)))), xlabel(strcat('Conventional Estim (n=',num2str(m),')')), axis image, set(gca,'xtick',[],'ytick',[])
subplot(1,5,4), imagesc(reshape(x_csl1,round(sqrt(n)),round(sqrt(n)))), xlabel(strcat('CS Reconstruction (n=',num2str(m),')')), axis image, set(gca,'xtick',[],'ytick',[])
subplot(1,5,5), imagesc(reshape(x_csl0,round(sqrt(n)),round(sqrt(n)))), xlabel(strcat('CS Reconstruction (n=',num2str(l),')')), axis image, set(gca,'xtick',[],'ytick',[])
colormap gray

% %
r = corr(A(:),x_gossl);
fprintf('Squared Correlation - Linear, High Sample: %5.4f\n',r^2)

r = corr(A(:),x_gossm);
fprintf('Squared Correlation - Linear, Low Sample: %5.4f\n',r^2)

% % r = corr(A(:),x_csl2);
% % fprintf('Squared Correlation - CS, Least Squares: %5.4f\n',r^2)

r = corr(A(:),x_csl1);
fprintf('Squared Correlation - CS, Zhang Passive: %5.4f\n',r^2)

r = corr(A(:),x_csl0);
fprintf('Squared Correlation - CS, Zhang Passive: %5.4f\n',r^2)

% end
